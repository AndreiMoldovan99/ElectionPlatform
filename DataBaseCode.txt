CREATE TABLE Voter
(
	vtr_id serial PRIMARY KEY,
	vtr_firstName character varying(20),
	vtr_lastName character varying(20),
	vtr_userName character varying(40),
	vtr_password character varying(40),
	vtr_cnp bigint
)

CREATE TABLE Party
(
	pa_id serial PRIMARY KEY,
	pa_partyName character varying(100),
	pa_partyDescription character varying(100)
)

CREATE TABLE Candidate
(
	ca_id serial PRIMARY KEY,
	ca_partyId integer,
	FOREIGN KEY (ca_partyId) REFERENCES Party (pa_id)
)

CREATE TABLE Vote
(
	vt_id serial PRIMARY KEY,
	vt_candidateId integer,
	vt_voterId integer, 
	FOREIGN KEY (vt_candidateId) REFERENCES Candidate (ca_id),
	FOREIGN KEY (vt_voterId) REFERENCES Voter (vtr_id)
)
